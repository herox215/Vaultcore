**1. Add flash highlight CM6 extension** to `src/components/Editor/EditorPane.svelte`:

Create the following CM6 primitives (either inline or in a separate `src/components/Editor/flashHighlight.ts` helper):

```typescript
import { StateEffect, StateField } from "@codemirror/state";
import { Decoration, type DecorationSet, EditorView } from "@codemirror/view";

const flashEffect = StateEffect.define<{ from: number; to: number } | null>();

const flashField = StateField.define<DecorationSet>({
  create: () => Decoration.none,
  update(deco, tr) {
    for (const e of tr.effects) {
      if (e.is(flashEffect)) {
        if (e.value === null) return Decoration.none;
        return Decoration.set([
          Decoration.mark({ class: "vc-flash-highlight" }).range(e.value.from, e.value.to),
        ]);
      }
    }
    return deco.map(tr.changes);
  },
  provide: (f) => EditorView.decorations.from(f),
});
```

Add `flashField` to the CM6 extensions list in EditorPane (the `buildExtensions` function or equivalent).

**2. Expose a `scrollToMatch` function** on EditorPane (or make it callable from the tab system):

```typescript
function scrollToMatch(view: EditorView, from: number, to: number) {
  // 1. Scroll to position
  view.dispatch({
    effects: EditorView.scrollIntoView(from, { y: "center" }),
  });
  // 2. Apply flash highlight
  view.dispatch({
    effects: flashEffect.of({ from, to }),
  });
  // 3. After 50ms, add fade-out class via requestAnimationFrame
  requestAnimationFrame(() => {
    const marks = view.dom.querySelectorAll(".vc-flash-highlight");
    marks.forEach((m) => m.classList.add("vc-flash-done"));
  });
  // 4. After 2600ms, remove the decoration
  setTimeout(() => {
    view.dispatch({ effects: flashEffect.of(null) });
  }, 2600);
}
```

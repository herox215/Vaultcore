Create `.github/actions/gen-vault/action.yml`:
```yaml
name: 'Generate or Cache 100k Vault'
description: 'Cache-first: restore a deterministic bench vault or regenerate via gen-vault.'
inputs:
  vault-path:
    description: 'Absolute path where the vault lives (input + output).'
    required: true
  count:
    description: 'Number of notes.'
    required: false
    default: '100000'
  seed:
    description: 'RNG seed.'
    required: false
    default: '42'
  cache-key-extra:
    description: 'Additional salt appended to the cache key.'
    required: false
    default: ''
outputs:
  cache-hit:
    description: 'Whether the cache was restored.'
    value: ${{ steps.cache.outputs.cache-hit }}
  version-hash:
    description: 'Value of `gen-vault --version`.'
    value: ${{ steps.version.outputs.hash }}
runs:
  using: 'composite'
  steps:
    - name: Get gen-vault version hash
      id: version
      shell: bash
      run: |
        cd src-tauri
        cargo build --release --bin gen-vault --quiet
        HASH=$(./target/release/gen-vault --version)
        echo "hash=$HASH" >> "$GITHUB_OUTPUT"

    - name: Restore vault cache
      id: cache
      uses: actions/cache@v4
      with:
        path: ${{ inputs.vault-path }}
        key: vault-${{ steps.version.outputs.hash }}-c${{ inputs.count }}-s${{ inputs.seed }}-${{ inputs.cache-key-extra }}

    - name: Generate vault if cache miss
      if: steps.cache.outputs.cache-hit != 'true'
      shell: bash
      run: |
        mkdir -p "${{ inputs.vault-path }}"
        ./src-tauri/target/release/gen-vault \
          --out  "${{ inputs.vault-path }}" \
          --count ${{ inputs.count }} \
          --seed  ${{ inputs.seed }}

    - name: Summarise
      shell: bash
      run: |
        echo "### gen-vault cache"       >> "$GITHUB_STEP_SUMMARY"
        echo "cache-hit: ${{ steps.cache.outputs.cache-hit }}" >> "$GITHUB_STEP_SUMMARY"
        echo "version-hash: \`${{ steps.version.outputs.hash }}\`" >> "$GITHUB_STEP_SUMMARY"
        echo "path: ${{ inputs.vault-path }}"       >> "$GITHUB_STEP_SUMMARY"
        echo "notes: $(find "${{ inputs.vault-path }}" -name '*.md' | wc -l)" >> "$GITHUB_STEP_SUMMARY"
```

name: CI
on:

    - cron: '0 3 * * *'     # nightly full matrix 03:00 UTC

jobs:
  # PR-fast path: ubuntu-only. Full matrix runs on schedule + push to main.
  rust:
    name: rust (${{ matrix.os }})
    strategy:
      fail-fast: false
      matrix:
        os: ${{ (github.event_name == 'pull_request') && fromJSON('["ubuntu-22.04"]') || fromJSON('["ubuntu-22.04","macos-13","macos-14","windows-latest"]') }}
    runs-on: ${{ matrix.os }}
    defaults:
      run:
        working-directory: src-tauri
    steps:
      - uses: actions/checkout@v4
      - name: Linux system deps
        if: runner.os == 'Linux'
        run: |
          sudo apt-get update
          sudo apt-get install -y libwebkit2gtk-4.1-dev libayatana-appindicator3-dev librsvg2-dev pkg-config build-essential curl wget libssl-dev
      - uses: dtolnay/rust-toolchain@stable
        with:
          components: clippy, rustfmt
      - uses: Swatinem/rust-cache@v2
        with:
          workspaces: src-tauri
      - run: cargo fmt --check
      - run: cargo clippy --workspace --all-targets -- -D warnings
      - run: cargo test --workspace --all-targets --quiet

  frontend:
    name: frontend (${{ matrix.os }})
    strategy:
      fail-fast: false
      matrix:
        os: ${{ (github.event_name == 'pull_request') && fromJSON('["ubuntu-22.04"]') || fromJSON('["ubuntu-22.04","macos-14","windows-latest"]') }}
    runs-on: ${{ matrix.os }}
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
        with:
          version: 9
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: 'pnpm'

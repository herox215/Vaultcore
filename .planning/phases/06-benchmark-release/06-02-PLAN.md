```rust
use std::path::PathBuf;
use std::sync::OnceLock;

/// Resolves the vault path for benches.
/// Returns Some(path) if VAULTCORE_BENCH_VAULT is set,
/// or if VAULTCORE_BENCH_MINI=1 (generates a 1000-note mini vault once per process).
/// Returns None if neither is set — caller should skip with a clear stderr message.
pub fn load_vault_path() -> Option<PathBuf> {
    if let Ok(p) = std::env::var("VAULTCORE_BENCH_VAULT") {
        let pb = PathBuf::from(p);
        if pb.is_dir() { return Some(pb); }
        eprintln!("[bench_common] VAULTCORE_BENCH_VAULT points to missing dir: {:?}", pb);
        return None;
    }
    if std::env::var("VAULTCORE_BENCH_MINI").as_deref() == Ok("1") {
        return Some(mini_vault());
    }
    None
}

/// Generate (or reuse) a mini 1000-note vault in a process-global tempdir.
fn mini_vault() -> PathBuf {
    static MINI: OnceLock<PathBuf> = OnceLock::new();
    MINI.get_or_init(|| {
        let dir = std::env::temp_dir().join("vaultcore-bench-mini");
        if dir.is_dir() && std::fs::read_dir(&dir).map(|d| d.count()).unwrap_or(0) > 0 {
            return dir;   // reuse
        }
        std::fs::create_dir_all(&dir).unwrap();
        let status = std::process::Command::new(env!("CARGO_BIN_EXE_gen-vault"))
            .args(["--out", dir.to_str().unwrap(), "--count", "1000", "--seed", "42"])
            .status().expect("gen-vault failed");
        assert!(status.success());
        dir
    }).clone()
}

/// Standard skip-message helper used by every bench when load_vault_path() is None.
pub fn skip_message(bench_name: &str) {
    eprintln!(
        "[{bench_name}] no bench vault. Set VAULTCORE_BENCH_VAULT=/path/to/100k-vault \
         or VAULTCORE_BENCH_MINI=1 to generate a 1000-note mini vault."
    );
}
```

mod bench_common;

use criterion::{criterion_group, criterion_main, Criterion, black_box};
use std::path::PathBuf;
use vaultcore_lib::indexer::IndexCoordinator;

fn perf_09_initial_indexing(c: &mut Criterion) {
    let Some(vault) = bench_common::load_vault_path() else {
        bench_common::skip_message("perf_09_initial_indexing");
        return;
    };
    let mut g = c.benchmark_group("perf_09_initial_indexing");
    g.sample_size(10);                         // heavy op — 10 samples
    g.measurement_time(std::time::Duration::from_secs(120));
    g.bench_function("build_from_disk", |b| {
        b.iter_with_setup(
            || tempfile::tempdir().unwrap(),    // fresh index dir per iter
            |idx_tmp| {
                // Build from cold: delete the `.vaultcore/` sidecar before open
                let _ = std::fs::remove_dir_all(vault.join(".vaultcore"));
                let coord = IndexCoordinator::open(&vault).expect("open_vault");
                black_box(coord);
            },
        );
    });
    g.finish();
}

fn perf_10_incremental_update(c: &mut Criterion) {
    // Pre-open the vault once; then measure applying a single-file update event.
    let Some(vault) = bench_common::load_vault_path() else {
        bench_common::skip_message("perf_10_incremental_update");
        return;
    };
    let coord = IndexCoordinator::open(&vault).expect("open_vault");
    // Pick one known .md file from the vault for the update path — scan once.
    let target = walkdir::WalkDir::new(&vault).into_iter()
        .filter_map(|e| e.ok())
        .find(|e| e.path().extension().and_then(|s| s.to_str()) == Some("md"))
        .map(|e| e.path().to_path_buf())
        .expect("at least one .md file");
    // Small stable edit: append a harmless marker, then revert in setup.
    let orig = std::fs::read(&target).unwrap();

    let mut g = c.benchmark_group("perf_10_incremental_update");
    g.bench_function("single_file_update", |b| {
        b.iter(|| {
            // Write a modified body
            let mut body = orig.clone();
            body.extend_from_slice(b"\n<!-- bench -->\n");
            std::fs::write(&target, &body).unwrap();
            // Invoke whatever the public incremental path is — discover from
            // src-tauri/src/indexer/mod.rs. Likely: coord.update_file(&target).
            // Adjust to real signature found via the read_first step.
            let _ = coord.update_file(&target);
            black_box(&target);
        });
    });
    g.finish();
    // Restore original
    std::fs::write(&target, &orig).unwrap();
}

criterion_group!(benches, perf_09_initial_indexing, perf_10_incremental_update);
criterion_main!(benches);

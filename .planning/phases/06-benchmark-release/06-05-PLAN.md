Create `scripts/audit/grep-telemetry.sh`:
```bash
#!/usr/bin/env bash
# Static source scan for network/telemetry API usage.
# Exit 0 = clean, exit 1 = offending matches found (printed).
set -eu

ROOT="$(cd "$(dirname "$0")/../.." && pwd)"
cd "$ROOT"

BLACKLIST=(
  # Rust HTTP clients
  'reqwest::' 'ureq::' 'hyper::Client' 'isahc::' 'attohttpc::' 'curl::'
  # Browser network APIs
  '\bfetch\s*\(' '\bXMLHttpRequest\b' '\bWebSocket\b' '\bEventSource\b' 'navigator\.sendBeacon'
  # Telemetry SDKs
  'sentry::' 'posthog' 'mixpanel' 'amplitude' 'datadog' 'segment' 'plausible'
  '@sentry/' '@posthog/' '\bga\s*\(' '\bgtag\s*\(' '_paq\.push'
)

# Paths to scan (product code only). Tests + the gen-vault bin + build artifacts excluded.
INCLUDE=(src src-tauri/src)
EXCLUDE_REGEX='(/target/|/node_modules/|/dist/|src-tauri/src/bin/|src-tauri/src/tests/|/tests/|\.min\.)'

found=0
tmp=$(mktemp)
for pat in "${BLACKLIST[@]}"; do
  grep -rInE "$pat" "${INCLUDE[@]}" 2>/dev/null | grep -vE "$EXCLUDE_REGEX" > "$tmp" || true
  if [ -s "$tmp" ]; then
    echo "=== Offending pattern: $pat ==="
    cat "$tmp"
    echo
    found=1
  fi
done
rm -f "$tmp"

# Also scan for raw URL literals outside localhost/loopback
URL_HITS=$(grep -rInE 'https?://[a-zA-Z0-9]' "${INCLUDE[@]}" 2>/dev/null \
  | grep -vE "$EXCLUDE_REGEX" \
  | grep -vE 'https?://(localhost|127\.0\.0\.1|\[::1\])(:[0-9]+)?' \
  | grep -vE '//' || true)
# The final //-filter drops lines that are clearly comments/docs. Refine over time.
if [ -n "$URL_HITS" ]; then
  echo "=== Raw URL literals outside localhost ==="
  echo "$URL_HITS"
  found=1
fi

if [ "$found" -eq 0 ]; then
  echo "[grep-telemetry] CLEAN — no network/telemetry patterns in product source."
  exit 0
else
  echo "[grep-telemetry] FAIL — offending matches above."
  exit 1
fi
```

Create `scripts/audit/capability-review.sh`:
```bash
#!/usr/bin/env bash
# Tauri capability file permission whitelist check.
# Reads every src-tauri/capabilities/*.json and verifies only allowed prefixes appear.
set -eu

ROOT="$(cd "$(dirname "$0")/../.." && pwd)"
CAPDIR="$ROOT/src-tauri/capabilities"

# Allowed permission prefixes (as strings or identifiers).
ALLOWED_REGEX='^(core:|dialog:|fs:)'

if [ ! -d "$CAPDIR" ]; then
  echo "[capability-review] capabilities dir not found: $CAPDIR" >&2
  exit 1
fi

fail=0
for f in "$CAPDIR"/*.json; do
  [ -e "$f" ] || continue
  echo "--- $f"
  # Extract all string values of key "permissions" (they can be strings or {identifier: "..."}).
  # We rely on Python for robust JSON; Python 3 is present in all CI runners and macOS.
  python3 - "$f" <<'PY'
import json, re, sys, os
p = sys.argv[1]
with open(p) as fh:
    doc = json.load(fh)

allowed_re = re.compile(r'^(core:|dialog:|fs:)')
perms = doc.get("permissions", [])
offenders = []
for entry in perms:
    if isinstance(entry, str):
        if not allowed_re.match(entry):
            offenders.append(entry)
    elif isinstance(entry, dict):
        ident = entry.get("identifier", "<no-identifier>")
        if not allowed_re.match(ident):
            offenders.append(ident)
    else:
        offenders.append(repr(entry))

if offenders:
    print("OFFENDERS:")
    for o in offenders:
        print(f"  - {o}")
    sys.exit(2)
else:
    print(f"OK ({len(perms)} permissions, all whitelisted)")
    sys.exit(0)
PY
  rc=$?
  if [ "$rc" -ne 0 ]; then
    fail=1
  fi
done

if [ "$fail" -eq 0 ]; then
  echo "[capability-review] CLEAN."
  exit 0
else
  echo "[capability-review] FAIL — offending identifiers above."
  exit 1
fi
```

Create `scripts/audit/network-trace.sh`:
```bash
#!/usr/bin/env bash
# Runtime strace-based network syscall capture.
# Linux only — macOS/Windows equivalents documented in scripts/audit/README.md.
set -eu

if [ "$(uname -s)" != "Linux" ]; then
  echo "[network-trace] SKIP — strace only available on Linux. Run manually on macOS/Windows per scripts/audit/README.md."
  exit 0
fi

if ! command -v strace >/dev/null 2>&1; then
  echo "[network-trace] strace not installed. On Ubuntu: sudo apt-get install -y strace"
  exit 1
fi

ROOT="$(cd "$(dirname "$0")/../.." && pwd)"
cd "$ROOT"

BIN="$ROOT/src-tauri/target/release/vaultcore"
if [ ! -x "$BIN" ]; then
  echo "[network-trace] building release binary first (this may take a few minutes)..."
  (cd "$ROOT" && pnpm install --frozen-lockfile && pnpm tauri build --no-bundle)
fi

TRACE="$(mktemp /tmp/vaultcore-strace.XXXXXX.log)"
TIMEOUT_SEC=30
echo "[network-trace] starting binary under strace for ${TIMEOUT_SEC}s..."

timeout --signal=TERM "$TIMEOUT_SEC" strace -f -e trace=%network -o "$TRACE" "$BIN" &
STRACE_PID=$!
wait "$STRACE_PID" || true

echo "[network-trace] strace log: $TRACE"

# Look for connect() to non-loopback addresses.
# sa_family=AF_INET + sin_addr NOT in 127.0.0.0/8 / ::1 = violation.
OFFENDERS=$(grep -E 'connect\(' "$TRACE" \
  | grep -vE 'sin_addr\("127\.' \
  | grep -vE 'sin_addr\(htonl\(INADDR_LOOPBACK' \
  | grep -vE 'sin6_addr="::1"' \
  | grep -vE 'sun_path=' \
  | grep -vE 'AF_NETLINK' \
  || true)

# Also flag sendto with non-loopback addrs.
SENDTO=$(grep -E 'sendto\(' "$TRACE" \
  | grep -vE '127\.0\.0\.1' \
  | grep -vE 'AF_UNIX' \
  | grep -vE 'AF_NETLINK' \
  || true)

if [ -n "$OFFENDERS" ] || [ -n "$SENDTO" ]; then
  echo "[network-trace] FAIL — non-loopback network syscalls detected:"
  [ -n "$OFFENDERS" ] && echo "--- connect() offenders ---" && echo "$OFFENDERS"
  [ -n "$SENDTO" ]    && echo "--- sendto() offenders ---" && echo "$SENDTO"
  echo
  echo "Full trace retained at: $TRACE"
  exit 1
fi

echo "[network-trace] CLEAN — no non-loopback network syscalls in ${TIMEOUT_SEC}s window."
rm -f "$TRACE"
exit 0
```

# Security Audit Scripts

Three complementary scripts enforce SEC-01/02/03 per CONTEXT.md.

All three exit 0 on clean, 1 on violation.

name: Security Audit
on:
  pull_request:
    branches: [main]

    - cron: '30 3 * * *'      # nightly 03:30 UTC (after CI)

jobs:
  static-scan:
    name: Static source scan + capability whitelist
    runs-on: ubuntu-22.04

      - name: Install Python 3

  cargo-audit:
    # Nightly + dispatch only (requires rust toolchain; skip on PR to keep PR fast).
    if: github.event_name != 'pull_request'
    name: cargo audit
    runs-on: ubuntu-22.04
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
        with: { workspaces: src-tauri }
      - name: Install cargo-audit
        run: cargo install cargo-audit --locked

        run: cargo audit --deny warnings

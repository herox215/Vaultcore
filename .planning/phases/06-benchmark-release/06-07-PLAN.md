```typescript
#!/usr/bin/env tsx
/**
 * Performance gate: reads criterion + frontend-perf artifacts, compares to budgets,
 * emits Markdown summary, exits 1 on any hard-miss.
 */
import { readFileSync, readdirSync, writeFileSync, existsSync, mkdirSync } from 'node:fs';
import { join } from 'node:path';
import { execSync } from 'node:child_process';

type Budget =
  | { kind: 'ms'; perf_id: string; label: string; budget_ms: number }
  | { kind: 'mb'; perf_id: string; label: string; budget_mb: number };

const BUDGETS: Budget[] = [
  { kind: 'ms', perf_id: 'perf_01', label: 'Cold start',            budget_ms: 3000 },
  { kind: 'ms', perf_id: 'perf_02', label: 'Warm start',            budget_ms: 5000 },
  { kind: 'ms', perf_id: 'perf_03', label: 'Open note',             budget_ms: 100 },
  { kind: 'ms', perf_id: 'perf_04', label: 'Keystroke latency',     budget_ms: 16 },
  { kind: 'ms', perf_id: 'perf_04_10k', label: 'Keystroke 10k-line', budget_ms: 32 },
  { kind: 'ms', perf_id: 'perf_05', label: 'Full-text search',      budget_ms: 50 },
  { kind: 'ms', perf_id: 'perf_06', label: 'Quick Switcher',        budget_ms: 10 },
  { kind: 'ms', perf_id: 'perf_07', label: 'Backlinks',             budget_ms: 20 },
  { kind: 'ms', perf_id: 'perf_08', label: 'Link autocomplete',     budget_ms: 10 },
  { kind: 'ms', perf_id: 'perf_09', label: 'Initial indexing',      budget_ms: 60000 },
  { kind: 'ms', perf_id: 'perf_10', label: 'Incremental update',    budget_ms: 5 },
  { kind: 'mb', perf_id: 'perf_11', label: 'RAM idle (proxy)',      budget_mb: 100 },
  { kind: 'mb', perf_id: 'perf_12', label: 'RAM active (proxy)',    budget_mb: 250 },
];

interface GateRow { perf_id: string; label: string; measured: string; budget: string; passed: boolean; soft: boolean; source: string; }

function args() {
  const argv = process.argv.slice(2);
  const get = (f: string, def = '') => { const i = argv.indexOf(f); return i >= 0 ? argv[i+1] : def; };
  return {
    rustDir: get('--rust-dir', 'src-tauri/target/criterion'),
    frontendDir: get('--frontend-dir', 'target/perf-frontend'),
    obsidianRef: get('--obsidian-ref', 'scripts/perf/obsidian-reference.json'),
    softMiss: get('--soft-miss', 'perf_11,perf_12').split(',').filter(Boolean),
    out: get('--out', 'target/perf-gate-result.json'),
  };
}

function readCriterionMs(dir: string, group: string): number | null {
  // estimates.json lives at <group>/<bench>/new/estimates.json. Aggregate: take the max mean across sub-benches for the group.
  const gpath = join(dir, group);
  if (!existsSync(gpath)) return null;
  let max_ns = 0;
  for (const sub of readdirSync(gpath, { withFileTypes: true })) {
    if (!sub.isDirectory() || sub.name === 'report') continue;
    const est = join(gpath, sub.name, 'new', 'estimates.json');
    if (!existsSync(est)) continue;
    try {
      const doc = JSON.parse(readFileSync(est, 'utf-8'));
      const ns = doc.mean?.point_estimate ?? doc.Mean?.point_estimate ?? 0;
      if (ns > max_ns) max_ns = ns;
    } catch { /* skip */ }
  }
  return max_ns > 0 ? max_ns / 1e6 : null;
}

function readFrontendMs(dir: string, perfId: string): { p95_ms: number | null; idle_delta_mb?: number; active_delta_mb?: number } {
  // One or more JSONs per perf_id possible (e.g. perf_04 + perf_04_10k). Take max p95.
  if (!existsSync(dir)) return { p95_ms: null };
  let max_p95 = 0;
  let idle_mb: number | undefined;
  let active_mb: number | undefined;
  for (const f of readdirSync(dir)) {
    if (!f.startsWith(perfId) || !f.endsWith('.json')) continue;
    const doc = JSON.parse(readFileSync(join(dir, f), 'utf-8'));
    if (typeof doc.p95_ms === 'number') max_p95 = Math.max(max_p95, doc.p95_ms);
    if (typeof doc.idle_delta_mb === 'number') idle_mb = doc.idle_delta_mb;
    if (typeof doc.active_delta_mb === 'number') active_mb = doc.active_delta_mb;
  }
  return { p95_ms: max_p95 > 0 ? max_p95 : null, idle_delta_mb: idle_mb, active_delta_mb: active_mb };
}

function gitSha(): string {
  try { return execSync('git rev-parse --short HEAD', { encoding: 'utf-8' }).trim(); } catch { return 'unknown'; }
}

function main() {
  const a = args();
  const rows: GateRow[] = [];
  let hardFail = false;

  for (const b of BUDGETS) {
    const soft = a.softMiss.includes(b.perf_id);
    let measured = 'missing';
    let passed = false;
    let source = '';
    if (b.kind === 'ms') {
      // Rust benches → perf_01/02/05/07/08/09/10
      const isRust = ['perf_01','perf_02','perf_05','perf_07','perf_08','perf_09','perf_10'].includes(b.perf_id);
      if (isRust) {
        const ms = readCriterionMs(a.rustDir, b.perf_id);
        if (ms !== null) { measured = `${ms.toFixed(2)} ms`; passed = ms <= b.budget_ms; source = 'criterion'; }
      } else {
        // Frontend perf id variants (perf_04 and perf_04_10k).
        const key = b.perf_id === 'perf_04_10k' ? 'perf_04' : b.perf_id;
        const { p95_ms } = readFrontendMs(a.frontendDir, key);
        if (p95_ms !== null) { measured = `${p95_ms.toFixed(2)} ms (p95)`; passed = p95_ms <= b.budget_ms; source = 'frontend'; }
      }
      rows.push({ perf_id: b.perf_id, label: b.label, measured, budget: `${b.budget_ms} ms`, passed, soft, source });
      if (!passed && !soft && measured !== 'missing') hardFail = true;
      if (measured === 'missing' && !soft) hardFail = true;
    } else {
      const { idle_delta_mb, active_delta_mb } = readFrontendMs(a.frontendDir, 'perf_11_12');
      const v = b.perf_id === 'perf_11' ? idle_delta_mb : active_delta_mb;
      if (typeof v === 'number') { measured = `${v.toFixed(1)} MB`; passed = v <= b.budget_mb; source = 'frontend-proxy'; }
      rows.push({ perf_id: b.perf_id, label: b.label, measured, budget: `${b.budget_mb} MB`, passed, soft, source });
      // RAM is soft-gated per this plan's committed decision.
    }
  }

  // Markdown summary
  let md = `# Perf Gate Result\n\n- **Git SHA:** \`${gitSha()}\`\n- **Time:** ${new Date().toISOString()}\n- **Hard-fail:** ${hardFail ? 'YES' : 'NO'}\n\n| Metric | Measured | Budget | Source | Status |\n|--------|----------|--------|--------|--------|\n`;
  for (const r of rows) {
    const status = r.measured === 'missing' ? (r.soft ? '⚠️ missing (soft)' : '❌ missing') :
                   r.passed ? '✅ pass' :
                   r.soft ? '⚠️ over (soft)' : '❌ over';
    md += `| ${r.perf_id} ${r.label} | ${r.measured} | ${r.budget} | ${r.source} | ${status} |\n`;
  }

  console.log(md);

  // JSON artifact
  const outDir = a.out.split('/').slice(0,-1).join('/') || '.';
  if (!existsSync(outDir)) mkdirSync(outDir, { recursive: true });
  writeFileSync(a.out, JSON.stringify({ git_sha: gitSha(), hard_fail: hardFail, rows }, null, 2));

  // GitHub Actions job summary
  if (process.env.GITHUB_STEP_SUMMARY) {
    const fs = require('fs');
    fs.appendFileSync(process.env.GITHUB_STEP_SUMMARY, md + '\n');
  }

  process.exit(hardFail ? 1 : 0);
}

main();
```

```typescript
#!/usr/bin/env tsx
/**
 * Render the VaultCore-vs-Obsidian comparison table from a gate result + reference file.
 * Produces Markdown on stdout per CONTEXT.md §Specific Ideas.
 */
import { readFileSync } from 'node:fs';

interface Row { perf_id: string; label: string; measured: string; budget: string; passed: boolean; soft: boolean; source: string; }
interface GateDoc { git_sha: string; hard_fail: boolean; rows: Row[] }
interface RefEntry { label: string; budget_ms?: number; obsidian_ms?: number | null; budget_mb?: number; obsidian_mb?: number | null; source_line_spec: string; note?: string; }

function args() {
  const argv = process.argv.slice(2);
  const get = (f: string, d = '') => { const i = argv.indexOf(f); return i >= 0 ? argv[i+1] : d; };
  return {
    gateResult: get('--gate-result', 'target/perf-gate-result.json'),
    obsRef:     get('--obsidian-ref', 'scripts/perf/obsidian-reference.json'),
  };
}

const a = args();
const gate: GateDoc = JSON.parse(readFileSync(a.gateResult, 'utf-8'));
const ref: Record<string, RefEntry> = JSON.parse(readFileSync(a.obsRef, 'utf-8'));

console.log(`# VaultCore vs. Obsidian Performance Comparison\n`);
console.log(`**Git SHA:** \`${gate.git_sha}\` · **Gate:** ${gate.hard_fail ? '❌ fail' : '✅ pass'}\n`);
console.log(`| Metric | VaultCore | Budget | Obsidian reference | Speedup | Source |`);
console.log(`|--------|-----------|--------|--------------------|---------|--------|`);

for (const row of gate.rows) {
  const r = ref[row.perf_id] ?? ref[row.perf_id.replace('_10k', '')];
  const obs = r?.obsidian_ms ?? r?.obsidian_mb ?? null;
  const obsStr = obs === null || obs === undefined ? 'n/a' : (r?.obsidian_ms ? `${obs} ms` : `${obs} MB`);
  // Speedup: obsidian / measured (best-effort parse)
  let speedup = 'n/a';
  if (obs && row.measured !== 'missing') {
    const num = parseFloat(row.measured);
    if (!Number.isNaN(num) && num > 0) speedup = `${(Number(obs) / num).toFixed(1)}×`;
  }
  console.log(`| ${r?.label ?? row.label} | ${row.measured} | ${row.budget} | ${obsStr} | ${speedup} | ${r?.source_line_spec ?? 'n/a'} |`);
}
console.log(`\n*Obsidian numbers are static references from spec §7 on a comparable 100k-note vault. VaultCore numbers are measured in this CI run.*`);
```

```typescript
import { existsSync, mkdirSync, writeFileSync } from 'node:fs';
import { join } from 'node:path';
import { execSync } from 'node:child_process';

export interface BenchResult {
  perf_id: string;
  label: string;
  samples: number;
  p50_ms: number;
  p95_ms: number;
  max_ms: number;
  budget_ms: number;
  passed: boolean;
  vault_size: number | null;
  git_sha: string;
}

const HRTIME_NS_PER_MS = 1_000_000n;

export interface BenchOpts {
  iterations?: number;       // default 50
  warmup?: number;           // default 5
  budgetMs: number;          // hard budget for `passed` flag
  perfId: string;            // e.g. "perf_03"
  vaultSize?: number;        // notes count if known
}

/** Run `fn` with warmup + measured iterations, collect ns durations, return aggregated stats. */
export async function bench(
  label: string,
  fn: () => void | Promise<void>,
  opts: BenchOpts,
): Promise<BenchResult> {
  const iterations = opts.iterations ?? 50;
  const warmup = opts.warmup ?? 5;
  for (let i = 0; i < warmup; i++) await fn();
  const samples_ns: bigint[] = [];
  for (let i = 0; i < iterations; i++) {
    const t0 = process.hrtime.bigint();
    await fn();
    samples_ns.push(process.hrtime.bigint() - t0);
  }
  const samples_ms = samples_ns.map(n => Number(n) / 1e6).sort((a, b) => a - b);
  const p = (q: number) => samples_ms[Math.min(samples_ms.length - 1, Math.floor(samples_ms.length * q))];
  const max_ms = samples_ms[samples_ms.length - 1];
  const result: BenchResult = {
    perf_id: opts.perfId,
    label,
    samples: iterations,
    p50_ms: +p(0.5).toFixed(3),
    p95_ms: +p(0.95).toFixed(3),
    max_ms: +max_ms.toFixed(3),
    budget_ms: opts.budgetMs,
    passed: p(0.95) <= opts.budgetMs,    // gate on p95
    vault_size: opts.vaultSize ?? null,
    git_sha: gitSha(),
  };
  return result;
}

export function persistResult(r: BenchResult): string {
  const outDir = process.env.VAULTCORE_PERF_OUT_DIR ?? 'target/perf-frontend';
  if (!existsSync(outDir)) mkdirSync(outDir, { recursive: true });
  const path = join(outDir, `${r.perf_id}.json`);
  writeFileSync(path, JSON.stringify(r, null, 2));
  return path;
}

export function requireBenchVault(): string | null {
  const v = process.env.VAULTCORE_BENCH_VAULT;
  if (v && existsSync(v)) return v;
  if (process.env.VAULTCORE_BENCH_MINI === '1') return ensureMiniVault();
  return null;
}

let MINI_PATH: string | null = null;
function ensureMiniVault(): string {
  if (MINI_PATH && existsSync(MINI_PATH)) return MINI_PATH;
  const dir = join(process.env.TMPDIR ?? '/tmp', 'vaultcore-perf-mini');
  if (!existsSync(dir)) mkdirSync(dir, { recursive: true });
  // Reuse if non-empty
  try {
    const entries = require('node:fs').readdirSync(dir);
    if (entries.length === 0) {
      execSync(`cargo run --quiet --release --manifest-path src-tauri/Cargo.toml --bin gen-vault -- --out ${dir} --count 1000 --seed 42`,
               { stdio: 'inherit' });
    }
  } catch {
    execSync(`cargo run --quiet --release --manifest-path src-tauri/Cargo.toml --bin gen-vault -- --out ${dir} --count 1000 --seed 42`,
             { stdio: 'inherit' });
  }
  MINI_PATH = dir;
  return dir;
}

function gitSha(): string {
  try { return execSync('git rev-parse --short HEAD', { encoding: 'utf-8' }).trim(); }
  catch { return 'unknown'; }
}

/** Convenience used by every perf spec to skip cleanly. */
export function skipUnlessVault(): { vaultPath: string } | null {
  const v = requireBenchVault();
  if (!v) {
    console.warn('[perf] no bench vault. Set VAULTCORE_BENCH_VAULT or VAULTCORE_BENCH_MINI=1');
    return null;
  }
  return { vaultPath: v };
}
```

```typescript
describe('PERF-03: Open Note < 100 ms', () => {
```

**`tests/perf/perf_11_12_ram.test.ts`** — measures Node-process RSS as a proxy for app idle/active RAM. Hard caveat documented.

```typescript
import { describe, it, expect } from 'vitest';
import { writeFileSync, mkdirSync, existsSync } from 'node:fs';
import { join } from 'node:path';
import { skipUnlessVault } from './setup';

const MB = 1024 * 1024;

describe('PERF-11/12: RAM (proxy via process.memoryUsage)', () => {
  const ctx = skipUnlessVault();

  it.runIf(ctx)('measures idle and active RSS deltas for trend tracking', async () => {
    if (global.gc) global.gc();
    const t0 = process.memoryUsage();

    // Idle: just import the IPC layer + a few stores, no editor/UI.
    await import('../../src/ipc/commands');
    if (global.gc) global.gc();
    await new Promise(r => setTimeout(r, 100));
    const t_idle = process.memoryUsage();

    // Active: open 5 EditorViews with 500-word docs (mimics 5-tab session).
    const { EditorState } = await import('@codemirror/state');
    const { EditorView } = await import('@codemirror/view');
    const { markdown } = await import('@codemirror/lang-markdown');

    const word = 'lorem ipsum dolor sit amet ';
    const views: any[] = [];
    for (let i = 0; i < 5; i++) {
      const dom = document.createElement('div');
      document.body.appendChild(dom);
      views.push(new EditorView({
        state: EditorState.create({ doc: word.repeat(100), extensions: [markdown()] }),
        parent: dom,
      }));
    }
    if (global.gc) global.gc();
    await new Promise(r => setTimeout(r, 100));
    const t_active = process.memoryUsage();

    views.forEach(v => v.destroy());

    const idle_mb  = (t_idle.rss   - t0.rss) / MB;
    const active_mb = (t_active.rss - t0.rss) / MB;

    // PROXY: jsdom + Node ≠ Tauri WebView. We persist the deltas for trend tracking.
    // Real RAM gating happens in Plan 06 24h soak via OS-level RSS measurement of the
    // built Tauri binary. Here we just record and assert non-pathological growth.
    const out = {
      perf_id: 'perf_11_12',
      label: 'ram_proxy',
      idle_delta_mb: +idle_mb.toFixed(2),
      active_delta_mb: +active_mb.toFixed(2),
      // PERF-11 budget = 100 MB total app RAM idle, PERF-12 = 250 MB active.
      // jsdom proxy is a SUBSET — we use it only for relative regression detection.
      // Soft-gate: idle delta should be < 50 MB (Node baseline ~30 MB), active < 150 MB.
      idle_soft_budget_mb: 50,
      active_soft_budget_mb: 150,
      passed: idle_mb < 50 && active_mb < 150,
      note: 'jsdom proxy; real RAM measured in Plan 06 soak via OS RSS of Tauri build',
    };
    const outDir = process.env.VAULTCORE_PERF_OUT_DIR ?? 'target/perf-frontend';
    if (!existsSync(outDir)) mkdirSync(outDir, { recursive: true });
    writeFileSync(join(outDir, 'perf_11_12.json'), JSON.stringify(out, null, 2));
    expect(out.passed, `idle=${idle_mb.toFixed(1)}MB active=${active_mb.toFixed(1)}MB`).toBe(true);
  });
});
```

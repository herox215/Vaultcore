    3. Define the CompletionSource:
       ```typescript
       export async function wikiLinkCompletionSource(
         ctx: CompletionContext
       ): Promise<CompletionResult | null> {
         // Match [[ followed by any non-] characters (the query typed so far)
         const match = ctx.matchBefore(/\[\[([^\]]*)/);
         if (!match) return null;

         // Don't trigger if there's already a ]] after the cursor on the same line
         // (prevents re-triggering inside completed links)
         const lineText = ctx.state.doc.lineAt(ctx.pos).text;
         const posInLine = ctx.pos - ctx.state.doc.lineAt(ctx.pos).from;
         const afterCursor = lineText.slice(posInLine);
         if (afterCursor.startsWith("]]")) return null;

         // Don't trigger if user is typing alias (after |)
         const innerText = match.text.slice(2); // strip [[
         if (innerText.includes("|")) return null;

         const query = innerText;

         // Fetch results from backend (nucleo fuzzy match, D-05)
         let results;
         try {
           results = await suggestLinks(query, 20);
         } catch {
           return null; // silently fail if backend unavailable
         }

         if (results.length === 0) {
           return {
             from: match.from + 2,
             options: [{
               label: "Keine Dateien gefunden",
               apply: query, // keep typed text
               type: "text",
             }],
             filter: false,
           };
         }

         return {
           from: match.from + 2,
           options: results.map((r) => ({
             label: basename(r.path),
             detail: r.path,  // relative path shown in grey (D-04)
             apply: `${basename(r.path)}]]`,  // insert name + closing brackets
             type: "file",
             boost: r.score,
           })),
           filter: false, // backend already filtered via nucleo
         };
       }
       ```

       Key design decisions:
       - `filter: false` because nucleo already does fuzzy filtering on the backend. CM6's default filter would re-filter the already-filtered results with a different algorithm.
       - `apply` includes `]]` so the user gets a complete `[[Filename]]`. The `[[` is already typed.
       - Alias support (D-06): if the user types `|` after selecting, the popup doesn't reopen because `innerText.includes("|")` returns true and the source returns null.

       ```typescript
       import { RangeSetBuilder } from "@codemirror/state";
       import { syntaxTree } from "@codemirror/language";
       import type { DecorationSet } from "@codemirror/view";
       ```

    2. Define the wiki-link regex (same as Rust side):
       ```typescript
       const WIKI_LINK_RE = /\[\[([^\]|]+?)(?:\|([^\]]*))?\]\]/g;
       ```

    3. Define `resolvedPaths: Set<string>` as a module-level mutable set. This is populated by `setResolvedPaths(paths: string[])` which the parent component calls after fetching from backend. This avoids async IPC inside the ViewPlugin update loop.

    4. Helper `isInsideCodeBlock(state: EditorState, pos: number): boolean`:
       ```typescript
       const node = syntaxTree(state).resolve(pos, 1);
       let cur = node;
       while (cur) {
         const name = cur.type.name;
         if (name === "FencedCode" || name === "CodeBlock" || name === "InlineCode") return true;
         cur = cur.parent;
       }
       return false;
       ```
       Note: lezer node names must be verified at runtime. If "FencedCode" is not correct, try logging `node.type.name` during development and adjust. The research assumption A1 flags this.

    5. Define `buildDecorations(view: EditorView): DecorationSet`:
       - Create `RangeSetBuilder<Decoration>`.
       - Get the document text: `view.state.doc.toString()`.
       - Iterate regex matches with `WIKI_LINK_RE.exec(text)`.
       - For each match:
         a. Check `isInsideCodeBlock(view.state, match.index)` — skip if true.
         b. Extract target (group 1), strip `.md` suffix.
         c. Determine if resolved: check `resolvedPaths` set for a path whose stem matches target (case-insensitive stem compare).
         d. Apply `Decoration.mark({ class: resolved ? "cm-wikilink-resolved" : "cm-wikilink-unresolved", attributes: { "data-wiki-target": target, "data-wiki-resolved": resolved ? "true" : "false" } })`.
         e. Add to builder: `builder.add(match.index, match.index + match[0].length, decoration)`.
       - CRITICAL: Collect all matches into array first, sort by `from` position, then add to RangeSetBuilder in order. RangeSetBuilder panics if ranges are out of order.
       - Reset regex lastIndex to 0 before use (global regex stateful).
       - Return `builder.finish()`.

    6. Define `wikiLinkPlugin` as ViewPlugin:
       ```typescript
       export const wikiLinkPlugin = ViewPlugin.fromClass(
         class {
           decorations: DecorationSet;
           constructor(view: EditorView) {
             this.decorations = buildDecorations(view);
           }
           update(update: ViewUpdate) {
             if (update.docChanged || update.viewportChanged) {
               this.decorations = buildDecorations(update.view);
             }
           }
         },
         {
           decorations: (v) => v.decorations,
           eventHandlers: {
             mousedown(event: MouseEvent, view: EditorView) {
               // Only handle Ctrl/Cmd+Click or plain click on decorated spans
               const target = event.target as HTMLElement;
               const wikiTarget = target.closest("[data-wiki-target]");
               if (!wikiTarget) return false;

               const linkTarget = wikiTarget.getAttribute("data-wiki-target");
               const isResolved = wikiTarget.getAttribute("data-wiki-resolved") === "true";
               if (!linkTarget) return false;

               event.preventDefault();
               event.stopPropagation();

               // Dispatch custom event for the Svelte layer to handle
               // (avoids importing store directly into CM6 extension)
               view.dom.dispatchEvent(
                 new CustomEvent("wiki-link-click", {
                   bubbles: true,
                   detail: { target: linkTarget, resolved: isResolved },
                 })
               );
               return true;
             },
           },
         }
       );
       ```
